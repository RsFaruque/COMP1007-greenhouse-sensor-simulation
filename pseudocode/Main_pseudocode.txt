        endProgram (boolean) = false

                     4. Add data readings
                     5. Delete data readings

            choice (int) <- getValidInputNumber(prompt, 1, 6)

            IF :
                CASE 1:
                    dataRange = "all"
                CASE 2:
                    dataRange = "zone"
                CASE 3:
                    dataRange = "sensor"

                    csvData.addData()
                CASE 5:
                    csvData.deleteData()
                CASE 6:
                    PRINT "GOODBYE"
                    endProgram = True

            prompt = "Select a statistic:
                      1. Total nmber of readings
                      2. Average value
                      3. Minimum value
                      4. Maximum value
                      5. Numer of readings outside safe range
                      6. Percentage of readings outside safe range

    END MAIN

    SUB MODULE:
        NAME: getValidInputNumber
        PARAMETERS: prompt, startingRange, endingRange
        OUTPUT: int
        LOGIC:
            choice (int)
            DO:
                PRINT prompt
                TRY:
                    choice <- TAKE INPUT FROM USER USING THE PROMPT "Choice?: "
                    IF choice < startingRange OR choice > endingRange:
                        PRINT "Invalid input. Input out of range"
                EXCEPT InputMismatchException
                    PRINT "Invalid input. Input must be a number"
            WHILE(choice < startingRange OR choice > endingRange )
            RETRUN choice
        END LOGIC
    END SUBMODULE

END
