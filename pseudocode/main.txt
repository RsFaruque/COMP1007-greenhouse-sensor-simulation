CLASS Main
    CREATE A Logger OBJECT CALLED logger WITH PATH TO FILE
    CREATE A SensorReadingRange OBJECT CALLED sensorRange

    BEGIN main
        state (String) = "main"
        dataRange (String) = ""
        endProgram (boolean) = false
        CREATE A Scanner OBJECT CALLED scanner
        CREATE A SensorData OBJECT CALLED csvData WITH PATH TO CSV

        WHILE NOT endProgram
            IF state == "main"
                DISPLAY USING logger.logAndDisplay THE PROMPT
                    "Welcome to the Smart greenhouse Monitoring System:
                    Please select an option:
                    1. Statistics for the entire greenhouse"
                    2. Statistics by zone
                    3. Statistics by sensor type
                    4. Add data readings
                    5. Delete data readings
                    6. Exit program"
                
                choice = GET VALID INPUT USING getValidInputInRange(scanner, 1, 6)

                IF 
                    CASE 1 : 
                        dataRange = "all"
                        state = "sub"
                    CASE 2:
                        dataRange = selectZone(scanner)
                        state = "sub"
                    CASE 3:
                        dataRange = selectSensorType(scanner)
                        state = "sub"
                    CASE 4:
                        addData(scanner, csvData)
                    CASE 5:
                        deleteData(scanner, csvData)
                    CASE 6:
                        enProgram = true

            ELSE IF state == "sub"
                DISPLAY USING logger.logAndDisplay THE PROMPT
                    "Select a statistic:
                    1. Total number of readings
                    2. Average value
                    3. Minimum value
                    4. Maximum value
                    5. Number of readings outside safe range
                    6. Percentage of readings outside safe range
                    7. All statistics
                    8. Go back to main menu"
                
                choice = GET VALID INTEGER USING getValidInputInRange(scanner, 1, 8)
                IF 
                    choice == CASE 1 : DISPLAY TOTAL READING COUNT USING subOp1TotalCount(csvData, dataRange) 
                    choice == CASE 2 : DISPLAY AVERAGE READING USING subOp2MeanVal(dataRange, csvData)
                    choice == CASE 3 : DISPLAY LOWEST READING USING subOp3MinVal(dataRange, csvData)
                    choice == CASE 4 : DISPLAY HIGHEST READING USING subOp4MaxVal(dataRange, csvData)
                    choice == CASE 5 : DISPLAY NUMBER OF UNSAFE READING USING subOp5UnsafeCount(dataRange, csvData)
                    choice == CASE 6 : DISPLAY PERCENTAGE OF UNSAFE READING USING subOp6UnsafePercentage(dataRange, csvData)
                    choice == CASE 7 : DISPLAY ALL THE STATS LISTED IN THIS SUB MENU USING subOp7AllStat(dataRange, csvData)
                    choice == CASE 8:
                        state = "main"
                        dataRange = ""
        END WHILE
    END main


    METHOD NAME: getValidInputInRange
    IMPORT: scanner, start (INTEGER), end (INTEGER)
    EXPORT: INTEGER
    BEGIN METHOD:
        WHILE true
            TRY
                DISPLAY USING logger.logAndDisplay "Enter your choice: "
                choice = USING scanner TAKE INTEGER INPUT

                IF choice >= start AND choice <= end:
                    EXPORT choice
                ELSE
                    DISPLAY USING logger.logAndDisplay THE PROMPT "Invalid input. Selection must be within range."
            CATCH ANY EXCEPTION
                DISPLAY USING logger.logAndDisplay "Invalid input. Must be an integer"
                CLEAR THE SCANNER BUFFER
        END WHILE
    END METHOD


    METHOD NAME: selectZone
    IMPORT: scanner (Scanner)
    EXPORT: zone (STRING)
    BEGIN METHOD:
        DISPLAY USING logger.logAndDisplay 
            "Select one of the following zones:
            1. ZoneA
            2. ZoneB
            3. ZoneC"
        
        choice = GET A VALID INTEGER USING getValidInputInRange(scanner, 1, 3)
        IF 
            choice == CASE 1: EXPORT "ZoneA"
            choice == CASE 2: EXPORT "ZoneB"
            choice == CASE 3: EXPORT "ZoneC"
    END METHOD
    
    METHOD NAME: selectSensorType
    IMPORT: scanner (scanner)
    EXPORT: sensortype (STRING)
    BEGIN METHOD:
        DISPLAY USING logger.logAndDisplay
            "Select one of the sensor types:
            1. temperature
            2. humidity
            3. soil moisture
            4. light" 
        
        choice = GET A VALID INTEGER USING getValidInputInRange(scanner, 1, 4)
        IF 
            choice == CASE 1: EXPORT "temperature"
            choice == CASE 2: EXPORT "humidity"
            choice == CASE 3: EXPORT "soilMoisture"
            choice == CASE 4: EXPORT "light"
    END METHOD


    METHOD NAME: subOp1TotalCount
    IMPORT: dataRange(STRING), csvData(SensorData)
    EXPORT: null
    BEGIN METHOD:
        DISPLAY USING logger.logAndDisplay
            "The total number of <dataRange> sensors are <csvData.getTotalcount(dataRange)>"
    END METHOD

    METHOD NAME: subOp2MeanVal
    IMPORT: dataRange(STRING), csvData (SensorData)
    Export: null
    BEGIN METHOD:
        
            

END CLASS
